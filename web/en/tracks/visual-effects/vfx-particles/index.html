IN THE REAL GAME

A particle's life

Loop the slice, advance position, add gravity, fade by lifetime, remove.

type particle struct{ x, y, vx, vy, life float64 }

for i := range ps { // Update
    ps[i].x += ps[i].vx
    ps[i].y += ps[i].vy
    ps[i].vy += gravity
    ps[i].life--
}
