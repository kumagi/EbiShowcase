This is the real mechanism the entry calls. The two labelled layers keep source and explanation traceable.

func (g *game) Update() error {
	if g.clear || g.over {
		if retryPressed() {
			*g = *newGame(g.lesson)
		}
		return nil
	}
	g.frames++
	g.readMovement()
	if attackPressed() && g.attack == 0 {
		g.attack = 18
	}
	if toolPressed() {
		g.tool = (g.tool + 1) % 3
		g.message = g.tr("Tool selected: ", "どうぐを えらんだ: ") + g.toolName()
	}
	if g.attack > 0 {
		g.attack--
		if g.attack == 10 {
			g.resolveAction()
		}
	}
	g.player.Tick()
	g.player.Pos.X = max(48, min(432, g.player.Pos.X))
	g.player.Pos.Y = max(108, min(520, g.player.Pos.Y))
	g.collect()
	g.updateEnemies()
	g.updateLesson()
	g.updateParticles()
	if g.shake > 0 {
		g.shake--
	}
	if g.flash > 0 {
		g.flash--
	}
	return nil
}
