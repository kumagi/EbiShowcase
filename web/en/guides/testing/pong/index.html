func (g *game) Update() error {
	g.movePlayer()
	g.moveCPU()
	g.bounceBall()

	switch lessonlogic.ExitScore(g.ballY, -20, screenHeight+20) {
	case 0: // 上に抜けたらプレイヤーの得点
		g.playerScore++
		g.serve(1)
	case 1: // 下に抜けたら CPU の得点
		g.cpuScore++
		g.serve(-1)
	}
	return nil
}

EXTRACTED / complete pure logic

package lessonlogic
func ExitScore(y, min, max float64) int {
	if y < min {
		return 0
	}
	if y > max {
		return 1
	}
	return -1
}

package lessonlogic

import "testing"

func TestExitScore(t *testing.T) {
	tests := []struct {
		name string
		y    float64
		want int
	}{
		{name: "inside", y: 360, want: -1},
		{name: "upper boundary stays", y: -20, want: -1},
		{name: "leaves top", y: -20.1, want: 0},
		{name: "lower boundary stays", y: 740, want: -1},
		{name: "leaves bottom", y: 740.1, want: 1},
	}
	for _, tt := range tests {
		t.Run(tt.name, func(t *testing.T) {
			if got := ExitScore(tt.y, -20, 740); got != tt.want {
				t.Fatalf("ExitScore(%v) = %d, want %d", tt.y, got, tt.want)
			}
		})
	}
}
