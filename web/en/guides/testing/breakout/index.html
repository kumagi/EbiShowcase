func (g *game) Update() error {
	g.movePaddle()
	g.bounceWorld()
	g.hitBricks()

	// ボールが下に落ちた
	if g.ballY > screenHeight+15 {
		var gameOver bool
		g.lives, gameOver = lessonlogic.SpendLife(g.lives)
		if gameOver {
			*g = *newGame()
		} else {
			g.serve()
		}
	}

	// ブロックを全部壊したらクリア → 最初から
	if g.aliveBrickCount() == 0 {
		*g = *newGame()
	}
	return nil
}

EXTRACTED / complete pure logic

internal/lessonlogic/core_updates.go
package lessonlogic
func SpendLife(lives int) (nextLives int, gameOver bool) {
	nextLives = lives - 1
	return nextLives, nextLives <= 0
}

internal/lessonlogic/core_updates_test.go
package lessonlogic

import "testing"

func TestSpendLife(t *testing.T) {
	testCases := []struct {
		name         string
		lives        int
		wantLives    int
		wantGameOver bool
	}{
		{name: "spending one of three lives continues game", lives: 3, wantLives: 2, wantGameOver: false},
		{name: "spending last life ends game", lives: 1, wantLives: 0, wantGameOver: true},
	}

	for _, tc := range testCases {
		t.Run(tc.name, func(t *testing.T) {
			gotLives, gotGameOver := SpendLife(tc.lives)

			if gotLives != tc.wantLives {
				t.Errorf("lives = %d, want %d", gotLives, tc.wantLives)
			}
			if gotGameOver != tc.wantGameOver {
				t.Errorf("gameOver = %v, want %v", gotGameOver, tc.wantGameOver)
			}
		})
	}
}

Naming

Top-level subjects are TestSpendLife. Every t.Run name states the premise and expected result in a sentence; there is no whenFoo_shouldBar convention.

AAA

Case values arrange the premise, one call acts, and comparisons assert only its result. Blank lines separate the three paragraphs; Arrange / Act / Assert comments do not label them.

DAMP

Meaningful case names and got / want values stay visible. Assert contains no search loop or switch, and multiple return values are checked separately so each failure says one thing.

Table-driven

These are repeated boundary inputs to the same function, so every row is one complete table-driven case. The outer for runs cases; it is not a search hidden inside Assert.
