BEFORE / Everything lives in Update

func (g *game) Update() error {
  for i := range g.runners {
    r := &g.runners[i]
    r.gauge += r.speed
    if r.gauge >= 1000 {
      r.gauge = 0
      r.ready = true
    }
  }
  return nil
}

AFTER / Separate the rule

func AdvanceGauge(gauge, speed, limit int) (int, bool) {
  next := gauge + speed
  if next >= limit {
    return 0, true
  }
  return next, false
}

// Update applies the rule to each runner.
r.gauge, ready = lessonlogic.AdvanceGauge(r.gauge, r.speed, 1000)
