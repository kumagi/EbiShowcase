IN THE REAL GAME

粒の一生

配列をまわして位置を進め、重力を足し、寿命で薄くして消します。

type particle struct{ x, y, vx, vy, life float64 }

next := ps[:0] // 生き残りを同じ配列へ詰め直す
for _, p := range ps { // Update
    p.x += p.vx
    p.y += p.vy
    p.vy += gravity
    p.life--
    if p.life > 0 {
        next = append(next, p)
    }
}
ps = next // 寿命0の粒はここで退場
